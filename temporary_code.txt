package com.wrx.test;

import android.app.Activity;
import android.graphics.Canvas;
import android.graphics.Rect;
import android.os.Bundle;
import android.support.v7.widget.LinearLayoutManager;
import android.support.v7.widget.RecyclerView;
import android.view.View;

import com.wrx.test.recycle.MyRecycleAdapter;

/**
 * Created by wuruixiong on 2016/6/24.
 */
public class LinearRecycleViewDemo extends Activity{

    private RecyclerView mHorRecyclerView;
    private RecyclerView mVerRecyclerView;
    private MyRecycleAdapter mHorAdapter;
    private MyRecycleAdapter mVerAdapter;
    private LinearLayoutManager mHorManager;
    private LinearLayoutManager mVerManager;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.linear_recycle_view_layout);
        mHorRecyclerView = (RecyclerView) findViewById(R.id.horizonal_linear_recycler);
        mVerRecyclerView =  (RecyclerView) findViewById(R.id.vertical_linear_recycler);

        mHorAdapter = new MyRecycleAdapter(this);
        mHorManager = new LinearLayoutManager(this);
        mVerAdapter = new MyRecycleAdapter(this);
        mVerManager = new LinearLayoutManager(this);

        mHorManager.setOrientation(LinearLayoutManager.HORIZONTAL);
        mHorRecyclerView.setLayoutManager(mHorManager);
        mHorRecyclerView.setAdapter(mHorAdapter);
        mVerManager.setOrientation(LinearLayoutManager.VERTICAL);
        mVerRecyclerView.setLayoutManager(mVerManager);
        mVerRecyclerView.setAdapter(mVerAdapter);

    }

    @Override
    protected void onDestroy() {
        super.onDestroy();
    }
}

package com.wrx.test.recycle;

import android.content.Context;
import android.support.v7.widget.RecyclerView;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.TextView;

import com.wrx.test.R;

import java.util.zip.Inflater;

/**
 * Created by wuruixiong on 2016/6/26.
 */
public class MyRecycleAdapter extends RecyclerView.Adapter<MyViewHolder>{

    private Context mContext;

    public MyRecycleAdapter(Context context) {
        super();
        mContext = context;
    }

    @Override
    public int getItemViewType(int position) {
        return super.getItemViewType(position);
    }

    @Override
    public int getItemCount() {
        return 10;
    }

    @Override
    public MyViewHolder onCreateViewHolder(ViewGroup parent, int viewType) {
        View view = LayoutInflater.from(mContext).inflate(R.layout.recycle_item, null);
        MyViewHolder myViewHolder = new MyViewHolder(view);
        return myViewHolder;
    }

    @Override
    public void onBindViewHolder(MyViewHolder holder, int position) {
        View view = holder.itemView;
        TextView textView = (TextView) view.findViewById(R.id.item_text);
        String text = "text";
        for (int i = 0; i < position; i++) {
            text = text + "text";
        }
        textView.setText(text);
    }
}
